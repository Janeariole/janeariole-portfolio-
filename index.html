-- Loan Processing Database Schema
CREATE TABLE dbo.LoanApplications (
    ApplicationID UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWSEQUENTIALID(),
    CustomerID INT NOT NULL FOREIGN KEY REFERENCES dbo.Customers(CustomerID),
    LoanAmount DECIMAL(15,2) NOT NULL CHECK (LoanAmount > 0),
    ApplicationDate DATETIME2(3) DEFAULT SYSDATETIME(),
    Status VARCHAR(20) NOT NULL DEFAULT 'Pending'
        CHECK (Status IN ('Pending', 'Approved', 'Rejected', 'UnderReview')),
    CreditScore INT,
    RiskCategory AS (
        CASE
            WHEN CreditScore >= 750 THEN 'A+'
            WHEN CreditScore BETWEEN 700 AND 749 THEN 'A'
            WHEN CreditScore BETWEEN 650 AND 699 THEN 'B'
            ELSE 'C'
        END
    ) PERSISTED,
    ProcessingTimeMinutes AS (
        DATEDIFF(MINUTE, ApplicationDate, 
            COALESCE(ApprovalDate, LastUpdatedDate, ApplicationDate))
    ),
    INDEX IX_Status_Date (Status, ApplicationDate),
    INDEX IX_Customer_Risk (CustomerID, RiskCategory)
);

-- Power Platform Integration View
CREATE VIEW vw_PowerApps_LoanDashboard
WITH SCHEMABINDING
AS
SELECT
    la.ApplicationID,
    la.CustomerID,
    c.FullName,
    la.LoanAmount,
    FORMAT(la.ApplicationDate, 'yyyy-MM-dd HH:mm:ss') AS SubmissionTime,
    la.Status,
    la.CreditScore,
    la.RiskCategory,
    la.ProcessingTimeMinutes,
    CASE
        WHEN la.LoanAmount < 10000 AND la.CreditScore >= 700 
            THEN 'Auto-Approve'
        WHEN la.LoanAmount BETWEEN 10000 AND 50000 AND la.CreditScore >= 650 
            THEN 'Manager Review'
        ELSE 'Committee Review'
    END AS RecommendedAction,
    COUNT(*) OVER(PARTITION BY la.CustomerID) AS CustomerApplicationCount
FROM dbo.LoanApplications la
INNER JOIN dbo.Customers c ON la.CustomerID = c.CustomerID
WHERE la.ApplicationDate >= DATEADD(DAY, -30, SYSDATETIME());
